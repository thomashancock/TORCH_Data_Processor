233 std::array<unsigned int,4> FileReader::readDataBlock(
234  std::unique_ptr<std::ifstream>& inputData
235 ) {
236  // Format is DCBA DCBA DCBA DCBA. Want AAAA BBBB CCCC DDDD
237  // Each number representes one byte
238  // Number gives order in which each byte will be read
239  // Slot D 0, 4, 8, 12
240  // Slot C 1, 5, 9, 13
241  // Slot B 2, 6, 10, 14
242  // Slot A 3, 7, 11, 15
243 
244  // Declare array to store block
245  std::array<unsigned int, 4> block = {{ 0 }};
246 
247  // mult is used to offset each byte to construct a 32 bit word
248  for (const auto& mult : { 0x01000000, 0x00010000, 0x00000100, 0x00000001 } ) {
249  // Iterate in reverse order to give D = 3, C = 2, B = 1, A = 0
250  for (int i = 3; i >= 0; i--) {
251  block[i] += mult * inputData->get();
252  }
253  }
254 
255  // block will be moved due to RVO
256  return block;
257 }
