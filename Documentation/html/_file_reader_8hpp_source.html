186 std::array<unsigned int,4> FileReader::readDataBlock(
187  std::unique_ptr<std::ifstream>& inputData
188 ) {
189  // Format is DCBA DCBA DCBA DCBA. Want AAAA BBBB CCCC DDDD
190  // Each number representes one byte
191  // Number gives order in which each byte will be read
192  // Slot D 0, 4, 8, 12
193  // Slot C 1, 5, 9, 13
194  // Slot B 2, 6, 10, 14
195  // Slot A 3, 7, 11, 15
196 
197  // Declare array to store block
198  std::array<unsigned int, 4> block = {{ 0 }};
199 
200  // mult is used to offset eact byte to construct a 32 bit word
201  for (auto& mult : { 0x01000000, 0x00010000, 0x00000100, 0x00000001 } ) {
202  // Iterate in reverse order to give D = 3, C = 2, B = 1, A = 0
203  for (int i = 3; i >= 0; i--) {
204  block[i] += mult * inputData->get();
205  }
206  }
207 
208  // block will be moved due to RVO
209  return block;
210 }
